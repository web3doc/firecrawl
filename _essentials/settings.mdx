每个Mintlify站点都需要一个包含核心配置设置的`mint.json`文件。了解更多关于[属性](#properties)的信息。

## 属性

<ResponseField name="name" type="string" required>
项目的名称。用于全局标题。

示例：`mintlify`

</ResponseField>

<ResponseField name="navigation" type="Navigation[]" required>
  包含该组内所有页面的组数组
  <Expandable title="导航">
    <ResponseField name="group" type="string">
    组的名称。

    示例：`设置`

    </ResponseField>
    <ResponseField name="pages" type="string[]">
    将用作页面的Markdown文件的相对路径。

    示例：`["customization", "page"]`

    </ResponseField>

  </Expandable>
</ResponseField>

<ResponseField name="logo" type="string or object">
  Logo图像的路径或包含“light”和“dark”模式logo图像路径的对象
  <Expandable title="Logo">
    <ResponseField name="light" type="string">
      亮模式下的logo路径
    </ResponseField>
    <ResponseField name="dark" type="string">
      暗模式下的logo路径
    </ResponseField>
    <ResponseField name="href" type="string" default="/">
      点击logo后链接到的地址
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="favicon" type="string">
  favicon图像的路径
</ResponseField>

<ResponseField name="colors" type="Colors">
  全局主题的十六进制颜色代码
  <Expandable title="颜色">
    <ResponseField name="primary" type="string" required>
      主颜色。通常用于高亮内容、章节标题、强调色，在亮模式下使用。
    </ResponseField>
    <ResponseField name="light" type="string">
      暗模式下的主颜色。通常用于高亮内容、章节标题、强调色，在暗模式下使用。
    </ResponseField>
    <ResponseField name="dark" type="string">
      重要按钮的主颜色。
    </ResponseField>
    <ResponseField name="background" type="object">
      亮模式和暗模式下的背景颜色
      <Expandable title="对象">
        <ResponseField name="light" type="string" required>
          亮模式下背景的十六进制颜色代码
        </ResponseField>
        <ResponseField name="dark" type="string" required>
          暗模式下背景的十六进制颜色代码
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarLinks" type="TopbarLink[]">
  要在顶部栏中包含的链接的“名称”和“URL”数组
  <Expandable title="顶部栏链接">
    <ResponseField name="name" type="string">
    按钮的名称。

    示例：`联系我们`
    </ResponseField>
    <ResponseField name="url" type="string">
    点击按钮后的URL。示例：`https://mintlify.com/contact`
    </ResponseField>

  </Expandable>
</ResponseField>

<ResponseField name="topbarCtaButton" type="Call to Action">
  <Expandable title="顶部栏行动号召">
    <ResponseField name="type" type={'"link" or "github"'} default="link">
    链接显示一个按钮。GitHub在提供的URL处显示仓库信息，包括GitHub星标的数量。
    </ResponseField>
    <ResponseField name="url" type="string">
    如果是“链接”：按钮链接到的地址。
    
    如果是“GitHub”：加载GitHub信息的仓库链接。
    </ResponseField>
    <ResponseField name="name" type="string">
    按钮内的文本。如果`type`是“链接”，则必须提供。
    </ResponseField>

  </Expandable>
</ResponseField>

<ResponseField name="versions" type="string[]">
  版本名称数组。仅当您想在导航栏中显示不同版本的文档时使用此属性。
</ResponseField>

<ResponseField name="anchors" type="Anchor[]">
  锚点数组，包括`icon`、`color`和`url`。
  <Expandable title="锚点">
    <ResponseField name="icon" type="string">
    用于突出锚点的[Font Awesome](https://fontawesome.com/search?s=brands%2Cduotone)图标。


    示例：`comments`
    </ResponseField>
    <ResponseField name="name" type="string">
    锚点标签的名称。

    示例：`社区`
    </ResponseField>
    <ResponseField name="url" type="string">
      标记哪些页面属于该锚点的URL起始部分。通常是放置页面的文件夹名称。
    </ResponseField>
    <ResponseField name="color" type="string">
      锚点图标背景的十六进制颜色。也可以是一个渐变，如果您传递一个包含`from`和`to`属性的对象，它们都是十六进制颜色。
    </ResponseField>
    <ResponseField name="version" type="string">
      用于隐藏锚点，直到选择了正确的文档版本。
    </ResponseField>
    <ResponseField name="isDefaultHidden" type="boolean" default="false">
      如果要在直接链接到文档内部之前隐藏锚点，请传递`true`。
    </ResponseField>
    <ResponseField name="iconType" default="duotone" type="string">
      可以是以下之一："brands", "duotone", "light", "sharp-solid", "solid", 或 "thin"
    </ResponseField>

  </Expandable>
</ResponseField>

<ResponseField name="topAnchor" type="Object">
  覆盖最顶部锚点的默认配置。
  <Expandable title="对象">
    <ResponseField name="name" default="Documentation" type="string">
      最顶部锚点的名称
    </ResponseField>
    <ResponseField name="icon" default="book-open" type="string">
      Font Awesome图标。
    </ResponseField>
    <ResponseField name="iconType" default="duotone" type="string">
      可以是以下之一："brands", "duotone", "light", "sharp-solid", "solid", 或 "thin"
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="tabs" type="Tabs[]">
  导航标签数组。
  <Expandable title="标签">
    <ResponseField name="name" type="string">
      标签的名称。
    </ResponseField>
    <ResponseField name="url" type="string">
      标记哪些页面属于该标签的URL起始部分。通常是放置页面的文件夹名称。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="api" type="API">
  API设置的配置。更多关于API页面的信息，请参阅[API组件](/api-playground/demo)。
  <Expandable title="API">
    <ResponseField name="baseUrl" type="string">
      所有API端点的基础URL。如果`baseUrl`是一个数组，它将启用多个基础URL选项，用户可以切换。
    </ResponseField>

    <ResponseField name="auth" type="Auth">
      <Expandable title="认证">
        <ResponseField name="method" type='"bearer" | "basic" | "key"'>
          所有API端点使用的认证策略。
        </ResponseField>
        <ResponseField name="name" type="string">
        在API游乐场中使用的认证参数的名称。
        
        如果方法是`basic`，格式应为`[usernameName]:[passwordName]`
        </ResponseField>
        <ResponseField name="inputPrefix" type="string">
        设计为认证输入字段前缀的默认值。
        
        例如，如果`inputPrefix`是`AuthKey`，则继承认证字段的默认输入结果为`AuthKey`。
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="playground" type="Playground">
      API游乐场的配置

      <Expandable title="游乐场">
        <ResponseField name="mode" default="show" type='"show" | "simple" | "hide"'>
          游乐场是否显示
