```python
from firecrawl import FirecrawlApp

# 初始化客户端
firecrawl = FirecrawlApp(api_key="your_api_key")

# 定义研究参数
params = {
    "maxDepth": 5,  # 研究迭代次数
    "timeLimit": 180,  # 时间限制（秒）
    "maxUrls": 15  # 分析的最大URL数量
}

# 开始实时更新的研究
def on_activity(activity):
    print(f"[{activity['type']}] {activity['message']}")

# 运行深度研究
results = firecrawl.deep_research(
    query="量子计算的最新发展是什么？",
    params=params,
    on_activity=on_activity
)

# 访问研究发现
print(f"最终分析：{results['data']['finalAnalysis']}")
print(f"来源：{len(results['data']['sources'])} 参考文献")
```
