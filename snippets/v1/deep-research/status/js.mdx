```js
import FirecrawlApp from 'firecrawl';

// 初始化客户端
const firecrawl = new FirecrawlApp({ apiKey: 'your_api_key' });

// 检查研究状态
const status = await firecrawl.checkDeepResearchStatus('job_id');

// 打印当前进度
console.log(`状态: ${status.status}`);
console.log(`进度: ${status.currentDepth}/${status.maxDepth} 次迭代`);

if (status.status === 'completed') {
  console.log(`最终分析: ${status.data.finalAnalysis}`);
  console.log(`找到的来源数量: ${status.data.sources.length}`);
}
```
