from pydantic import BaseModel

class ExtractSchema(BaseModel):
    company_mission: str

# 定义要提取数据的URL
urls = [
    'https://firecrawl.dev/*'
]

# 定义提取提示
prompt = '从页面中提取公司使命。'

# 执行提取操作
scrape_result = app.extract(urls, {
    'prompt': prompt,
    'schema': ExtractSchema.model_json_schema(),
})

# 检查提取是否成功
if not scrape_result['success']:
    raise Exception(f"提取失败：{scrape_result['error']}")

# 打印提取的数据
print(scrape_result['data'])
