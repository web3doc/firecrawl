```js Node
import FirecrawlApp from '@mendable/firecrawl-js';

const app = new FirecrawlApp({
  apiKey: 'fc-YOUR_API_KEY',
});

const scrapeResult = await app.extract(['https://nextbase.com/dash-cams/622gw-dash-cam'], {
  prompt: '提取关于最佳行车记录仪的详细信息，包括价格、功能、优缺点和评论。',
  enableWebSearch: true, // 启用网络搜索以获得更好的上下文
});

if (!scrapeResult.success) {
  throw new Error(`抓取失败: ${scrapeResult.error}`);
}

console.log(scrapeResult.data);
```
